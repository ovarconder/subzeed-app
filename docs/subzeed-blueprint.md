## 1. MARKETING & SUBSCRIPTION TIER DESIGN (Anniversary Billing & Flexible Renewal)
ระบบนับรอบบิลแบบ Anniversary Billing (สมัครวันไหนนับวันนั้นไป 30 วัน) โดยมีกฎเหล็กในการคำนวณโควตาและงานบัญชีดังนี้:
*   **ใช้หมดก่อนครบรอบเดือน:** ลูกค้าต้องกดเติมเงินใหม่ตามราคาแพ็กเกจ โดยระบบจะรีเซ็ตวันหมดอายุขยับออกไปอีก 30 วันนับจากวันที่จ่ายก้อนใหม่ทันที
*   **ครบเดือนใช้ไม่หมด:** ตัดนาทีคงเหลือทิ้งทันที ไม่มีการทบไปรอบเดือนถัดไป (ยกยอดโควตากลับเป็นศูนย์เพื่อลุยรอบบิลใหม่)
*   **ยังใช้ไม่หมด แต่กดเติมก่อน (Renew Early):** ระบบจะคำนวณแบบใจดี โดยนำจำนวนนาทีคงเหลือเก่าที่ยังใช้ไม่หมด ไปบวกสมทบ (ทบเพิ่ม) เข้ากับจำนวนนาทีเต็มของแพ็กเกจใหม่ทันที และขยับวันหมดอายุของรอบบิลออกไปอีก 30 วันเต็มจากวินาทีที่กดยืนยันชำระเงินสำเร็จ
*   **Usage & Reports:** มีระบบแสดงสถานะการใช้งานแบบเรียลไทม์ (นาทีทั้งหมด / นาทีที่ใช้ไป) และเก็บ Log ประวัติอย่างละเอียดเพื่อให้ลูกค้าดึงรายงานออกไปส่งให้ฝ่ายบัญชีของบริษัทได้

## 2. CORE ARCHITECTURE & TECH STACK (Lean & Cost Control 100%)
*   **Frontend & Hosting:** Next.js (App Router) ดีพล็อยอยู่บน Vercel ภายใต้โฟลเดอร์ `/subzeed`
*   **Backend & Auth:** Supabase (จัดการระบบ Auth, PostgreSQL Database และจัดการสิทธิ์แบบ Row Level Security - RLS)
*   **Payment Gateway:** Stripe (ดักจับ Event การชำระเงินผ่านระบบ Stripe Webhooks และใช้ Stripe Customer Portal ให้ลูกค้าจัดการเรื่องบิลเพื่อลดการเขียนโค้ดหน้าบ้าน)
*   **State Management:** Zustand (จัดการ Array ออบเจกต์ของซับไตเติลฝั่งหน้าบ้าน เพื่อความ Reactive)
*   **STT Engine:** OpenAI Whisper (Large-v3) API ช่วง MVP 
*   **AI Smart Engine:** Gemini Flash API (ผ่านคีย์จาก Google AI Studio) สำหรับฟีเจอร์กรองและพิสูจน์อักษรคำศัพท์แบรนด์
*   **Vercel Optimization (Client-side Extraction):** บายพาสข้อจำกัด Serverless Timeout ของ Vercel ด้วยการดึงแทร็กเสียงออกจากวิดีโอ (Audio Extraction) บนเบราว์เซอร์ของลูกค้าโดยตรงผ่าน Web Audio API ข้อมูลที่ยิงผ่านเน็ตจึงมีแค่ไฟล์เสียงบีบอัดจิ๋ว (Mono 64kbps MP3) ขนาดไม่กี่ MB ส่งให้ Whisper ประมวลผลได้เร็วขึ้น 10 เท่า และค่าแบนด์วิธเซิร์ฟเวอร์เป็น 0 บาท
*   **Client-side Rendering:** ซ้อนเลเยอร์ HTML5 Canvas ครอบบนวิดีโอหน้าบ้านดึงฟอนต์เข้า Memory แล้วใช้อัดวิดีโอฝังซับ (Hardsub) ออกมาเป็นไฟล์สำเร็จผ่านเบราว์เซอร์ลูกค้าโดยตรง (MediaRecorder / WebCodecs) ค่า Server ในการเรนเดอร์เป็น 0 บาทตลอดชีพ

## 3. CORE DATABASE SCHEMA (Supabase / PostgreSQL)

```sql
CREATE TYPE subscription_tier AS ENUM ('free', 'basic', 'premium', 'business_starter', 'business_pro');
CREATE TYPE billing_action_type AS ENUM ('subscribe', 'renew_early', 'recurring', 'cancel', 'refund');
CREATE TYPE quota_log_type AS ENUM ('stt_transcription', 'renew_reset', 'admin_adjustment');

-- 1. ตารางโปรไฟล์ผู้ใช้ (รองรับระบบ Team Workspace แชร์โควตานาทีร่วมกันในองค์กร)
CREATE TABLE profiles (
    id UUID REFERENCES auth.users ON DELETE CASCADE PRIMARY KEY,
    email TEXT UNIQUE NOT NULL,
    phone_number TEXT,
    tier subscription_tier DEFAULT 'free' NOT NULL,
    quota_minutes_total NUMERIC(10, 2) DEFAULT 20.00 NOT NULL,
    quota_minutes_used NUMERIC(10, 2) DEFAULT 0.00 NOT NULL,
    billing_cycle_start TIMESTAMPTZ DEFAULT NOW() NOT NULL,
    billing_cycle_end TIMESTAMPTZ DEFAULT (NOW() + INTERVAL '1 month') NOT NULL,
    
    -- [Business Feature] ถ้าเป็น Member ในทีม ฟิลด์นี้จะเก็บ UUID ของ Owner คนที่จ่ายเงินเพื่อไปหักนาทีจากถังกลาง
    workspace_owner_id UUID REFERENCES profiles(id) ON DELETE SET NULL, 
    
    created_at TIMESTAMPTZ DEFAULT NOW() NOT NULL,
    updated_at TIMESTAMPTZ DEFAULT NOW() NOT NULL
);

-- 2. ตารางเก็บโปรเจกต์และก้อนซับไตเติล (รองรับสไตล์และการย้ายตำแหน่งของซับไตเติล)
CREATE TABLE projects (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    user_id UUID REFERENCES profiles(id) ON DELETE CASCADE NOT NULL,
    title TEXT DEFAULT 'วิดีโอไม่มีชื่อ' NOT NULL,
    video_url TEXT,
    duration_seconds NUMERIC(10, 2) DEFAULT 0.00 NOT NULL,
    
    -- โครงสร้างอาร์เรย์ซับ: [{"id": "sub-1", "start": 1.2, "end": 3.5, "text": "สวัสดี", "position": "bottom", "y_offset": 90}]
    subtitles JSONB DEFAULT '[]'::jsonb NOT NULL, 
    
    is_client_review_enabled BOOLEAN DEFAULT FALSE NOT NULL,
    review_token TEXT UNIQUE,
    created_at TIMESTAMPTZ DEFAULT NOW() NOT NULL,
    updated_at TIMESTAMPTZ DEFAULT NOW() NOT NULL
);

-- 3. ตารางประวัติธุรกรรม (ใช้สำหรับทำประวัติการชำระเงินและเจนเอกสารใบเสร็จ/ใบกำกับภาษี)
CREATE TABLE billing_history (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    user_id UUID REFERENCES profiles(id) ON DELETE CASCADE NOT NULL,
    action_type billing_action_type NOT NULL, -- subscribe, renew_early, recurring
    previous_tier subscription_tier NOT NULL,
    new_tier subscription_tier NOT NULL,
    amount_thb NUMERIC(10, 2) NOT NULL,
    invoice_number TEXT UNIQUE, -- รหัสรันเลขที่เอกสารบัญชี
    billing_cycle_start TIMESTAMPTZ NOT NULL,
    billing_cycle_end TIMESTAMPTZ NOT NULL,
    payment_status TEXT DEFAULT 'success' NOT NULL,
    created_at TIMESTAMPTZ DEFAULT NOW() NOT NULL
);

-- 4. ตารางบันทึกกิจกรรมโควตาอย่างละเอียดเพื่อส่งเป็น Report ให้ฝ่ายบัญชีของลูกค้าตรวจ
CREATE TABLE quota_activity_logs (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    user_id UUID REFERENCES profiles(id) ON DELETE CASCADE NOT NULL,
    project_id UUID REFERENCES projects(id) ON DELETE SET NULL,
    log_type quota_log_type NOT NULL,
    minutes_changed NUMERIC(10, 2) NOT NULL, -- บันทึกนาทีที่ใช้จริงแบบละเอียด เช่น -12.45 หรือ +2500.00
    quota_minutes_used_snapshot NUMERIC(10, 2) NOT NULL,
    description TEXT, -- เช่น "ถอดความคลิป 'Review_Final.mp4'" หรือ "ทบยอดเก่าและเติมโควตาใหม่"
    created_at TIMESTAMPTZ DEFAULT NOW() NOT NULL
);
```

## 4. FINANCIAL & INVOICE COMPLIANCE (ยุทธวิธีฝั่งบัญชีช่วงเริ่มต้น)
*   ช่วงยังไม่จด VAT: ระบบจะเจนไฟล์ PDF ผ่านหลังบ้านด้วยก้อนไลบรารีเบาๆ (เช่น pdfkit) ออกมาเป็น "ใบเสร็จรับเงิน (Receipt)" โดยแสดงเลขประจำตัวผู้เสียภาษี 13 หลักของฝั่งเรา แต่จะไม่มีการคำนวณแยกแถวภาษีมูลค่าเพิ่ม 7% และระบุหมายเหตุท้ายเอกสารสั้นๆ วางโครงสร้างไฟล์เผื่อไว้ให้กดดาวน์โหลดผ่านทางหน้า Dashboard ประวัติชำระเงินของลูกค้า

*   ช่วงจด VAT แล้วในอนาคต: ทำการเปิดสวิตช์ใน Code Logic หลังบ้านเพื่อแยกแสดงแถวภาษีมูลค่าเพิ่ม (VAT 7%) และเปลี่ยนชื่อหัวกระดาษเอกสารเป็น "ใบเสร็จรับเงิน / ใบกำกับภาษี (Receipt / Tax Invoice)" ได้อย่างสมบูรณ์ทันทีโดยไม่กระทบโครงสร้างฐานข้อมูลเดิม
